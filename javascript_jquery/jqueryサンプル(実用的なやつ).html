$('a>img').each(function(i,e){
  var off = $(e).attr('src');
  var on = off.replace(/^(.+)-off(\.[^\.]+)$/, '$1-on$2');
  $(e).hover(function(){
    $(this).attr('src',on);
  },function(){
    $(this).attr('src',off);
  });
});

$('#back a').click(goTop);
$('.scrollTop').click(goTop);
function goTop(){
  $('body,html').animate({  // bodyだけ、htmlだけでも効くけどブラウザによるので注意
    scrollTop:0
  },500);
  return false;
}

$('a[href^="#"]').click(function(){
  var href = $(this).attr('href');
  var pos = $(href).offset().top;
  $('body,html').animate({scrollTop:pos},1000);
  return false;
});

スクロールしたらポップアップ表示


.scroll_show_up{
  /* 表示する前の状態を設定 */
  position: relative;
  bottom: -100px;
  opacity: 0;
  transition: all 0.6s;
}
.scroll_show_up.show{
  /* 表示するときの動きを設定 */
  bottom: 0;
  opacity: 1;
}

.scroll_show_right{
  position: relative;
  left: -100px;
  opacity: 0;
  transition: all 0.6s;
}
.scroll_show_right.show{
  left: 0;
  opacity: 1;
}

var st = $(this).scrollTop();
// 以下のクラスの要素全てに対して処理を行う
$('.scroll_show_up, .scroll_show_right').each(function(){
  var elm = $(this);
  var elmpos = elm.offset().top;
  var wh = $(window).height();

  // スクロール量が対象要素のtopからの位置+画面の1/4になったらshowクラス付与
  // *画面に対象要素が1/4見えたらshow付与
  // show時の動きはcssにて設定。removeClassを消すと出しっぱなしになる。
  if ( st > elmpos - wh + wh /4){
    elm.addClass('show');
  }else{
    elm.removeClass('show');
  }
});

// ハンバーガーメニュー
$('.Toggle').click(function(){
  $(this).toggleClass('active');
  // .Toggleにactiveクラスが付いているかで分岐
  if($(this).hasClass('active')){
    $('.NavMenu').addClass('active');
  }else{
    $('.NavMenu').removeClass('active');
  }
});

/* ハンバーガーメニュー用css */
/*ナビメニューのスタイルを指定*/
nav.NavMenu{
  position: fixed; /*表示位置を固定*/
  z-index: 2; /*重ね順を変更*/
  top: 0; /*表示位置を指定*/
  left: 0; /*表示位置を指定*/
  background: #fff;/*背景を白にする*/
  color: #000; /*文字色を黒にする*/
  text-align: center; /*テキストを中央揃え*/
  width: 100%; /*全幅表示*/
  transform: translateY(-100%); /*ナビを上に隠す*/
  transition: all 0.6s; /*アニメーションの時間を指定*/
}
nav.NavMenu ul{
  background: #ccc; /*背景をグレーにする*/
  width: 100%;
  margin: 0 auto;
  padding: 0;
}
nav.NavMenu ul li{
  font-size: 1.1em;
  list-style-type: none;
  padding: 0;
  width: 100%;
  border-bottom: 1px dotted #333;
}
nav.NavMenu ul li:last-child{
  padding-bottom: 0;
  border-bottom: none; /*最後のメニュー項目のみ下線を消す*/

nav.NavMenu ul li a{
  display: block; /*クリックできる領域を広げる*/

  padding: 1em 0;
